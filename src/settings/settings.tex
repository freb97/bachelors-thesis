% Caption cite command
\newcommand{\captioncite}[3][]{%

            \fontsize{8pt}{6pt}\selectfont

        \endgroup
        \caption{#2}
    }{
        \begingroup
            \fontsize{8pt}{6pt}\selectfont
            Quelle: #1 \textpartcite{#2}
        \endgroup
        \caption{#3}
    }
